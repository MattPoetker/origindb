cmake_minimum_required(VERSION 3.20)
project(InstantDB VERSION 0.1.0 LANGUAGES CXX C)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)

# Build type
if(NOT CMAKE_BUILD_TYPE)
    set(CMAKE_BUILD_TYPE "Debug" CACHE STRING "Build type" FORCE)
endif()

# Options
option(BUILD_TESTS "Build unit tests" ON)
option(BUILD_BENCHMARKS "Build benchmarks" OFF)
option(ENABLE_ASAN "Enable Address Sanitizer" OFF)
option(ENABLE_TSAN "Enable Thread Sanitizer" OFF)

# Compiler flags
if(CMAKE_CXX_COMPILER_ID MATCHES "GNU|Clang")
    add_compile_options(-Wall -Wextra -Wpedantic)
    if(CMAKE_BUILD_TYPE STREQUAL "Debug")
        add_compile_options(-g -O0)
    elseif(CMAKE_BUILD_TYPE STREQUAL "Release")
        add_compile_options(-O3 -DNDEBUG)
    endif()
endif()

# Sanitizers
if(ENABLE_ASAN)
    add_compile_options(-fsanitize=address -fno-omit-frame-pointer)
    add_link_options(-fsanitize=address)
endif()

if(ENABLE_TSAN)
    add_compile_options(-fsanitize=thread -fno-omit-frame-pointer)
    add_link_options(-fsanitize=thread)
endif()

# Find packages
find_package(Threads REQUIRED)
find_package(Protobuf REQUIRED)

# Try to find spdlog, fmt, boost - if not found, fetch them
find_package(spdlog QUIET)
if(NOT spdlog_FOUND)
    message(STATUS "spdlog not found, will fetch from source")
endif()

find_package(fmt QUIET)
if(NOT fmt_FOUND)
    message(STATUS "fmt not found, will fetch from source")
endif()

find_package(Boost QUIET COMPONENTS system filesystem thread)
if(NOT Boost_FOUND)
    message(STATUS "Boost not found, will fetch from source")
endif()

# Third-party dependencies
include(FetchContent)

# JSON library
FetchContent_Declare(
    json
    GIT_REPOSITORY https://github.com/nlohmann/json.git
    GIT_TAG v3.11.2
)

# spdlog if not found
if(NOT spdlog_FOUND)
FetchContent_Declare(
    spdlog
    GIT_REPOSITORY https://github.com/gabime/spdlog.git
    GIT_TAG v1.12.0
)
endif()

# fmt if not found
if(NOT fmt_FOUND)
FetchContent_Declare(
    fmt
    GIT_REPOSITORY https://github.com/fmtlib/fmt.git
    GIT_TAG 10.1.1
)
endif()

# gRPC (always fetch for simplicity)
FetchContent_Declare(
    grpc
    GIT_REPOSITORY https://github.com/grpc/grpc.git
    GIT_TAG v1.60.0
    GIT_SUBMODULES_RECURSE TRUE
)

# NuRaft for consensus
FetchContent_Declare(
    nuraft
    GIT_REPOSITORY https://github.com/eBay/NuRaft.git
    GIT_TAG v1.3.0
)

# WebSocketPP for WebSocket support
FetchContent_Declare(
    websocketpp
    GIT_REPOSITORY https://github.com/zaphoyd/websocketpp.git
    GIT_TAG 0.8.2
)

# Wasmtime C++ API
FetchContent_Declare(
    wasmtime
    URL https://github.com/bytecodealliance/wasmtime/releases/download/v15.0.0/wasmtime-v15.0.0-x86_64-linux-c-api.tar.xz
    DOWNLOAD_EXTRACT_TIMESTAMP TRUE
)

FetchContent_MakeAvailable(json nuraft websocketpp)

# Wasmtime special handling
FetchContent_GetProperties(wasmtime)
if(NOT wasmtime_POPULATED)
    FetchContent_Populate(wasmtime)
    set(WASMTIME_INCLUDE_DIR ${wasmtime_SOURCE_DIR}/include)
    set(WASMTIME_LIB_DIR ${wasmtime_SOURCE_DIR}/lib)
endif()

# Include directories
include_directories(
    ${CMAKE_SOURCE_DIR}/include
    ${CMAKE_BINARY_DIR}/proto
    ${WASMTIME_INCLUDE_DIR}
)

# Proto files
file(GLOB PROTO_FILES "${CMAKE_SOURCE_DIR}/proto/*.proto")
protobuf_generate_cpp(PROTO_SRCS PROTO_HDRS ${PROTO_FILES})
grpc_generate_cpp(GRPC_SRCS GRPC_HDRS ${CMAKE_BINARY_DIR}/proto ${PROTO_FILES})

# Create proto library
add_library(proto_lib STATIC ${PROTO_SRCS} ${GRPC_SRCS})
target_link_libraries(proto_lib
    protobuf::libprotobuf
    gRPC::grpc++
)

# Source files
file(GLOB_RECURSE STORAGE_SRCS "src/storage/*.cpp")
file(GLOB_RECURSE RAFT_SRCS "src/raft/*.cpp")
file(GLOB_RECURSE SQL_SRCS "src/sql/*.cpp")
file(GLOB_RECURSE WASM_SRCS "src/wasm/*.cpp")
file(GLOB_RECURSE MODULE_SRCS "src/modules/*.cpp")
file(GLOB_RECURSE NET_SRCS "src/net/*.cpp")
file(GLOB_RECURSE CHANGEFEED_SRCS "src/changefeed/*.cpp")
file(GLOB_RECURSE ADMIN_SRCS "src/admin/*.cpp")
file(GLOB_RECURSE COMMON_SRCS "src/common/*.cpp")

# Libraries
add_library(storage STATIC ${STORAGE_SRCS})
target_link_libraries(storage
    spdlog::spdlog
    fmt::fmt
    Threads::Threads
)

add_library(raft_consensus STATIC ${RAFT_SRCS})
target_link_libraries(raft_consensus
    nuraft
    storage
    proto_lib
    spdlog::spdlog
)

add_library(sql_engine STATIC ${SQL_SRCS})
target_link_libraries(sql_engine
    storage
    spdlog::spdlog
)

add_library(wasm_host STATIC ${WASM_SRCS})
target_link_libraries(wasm_host
    ${WASMTIME_LIB_DIR}/libwasmtime.a
    storage
    spdlog::spdlog
)

add_library(module_store STATIC ${MODULE_SRCS})
target_link_libraries(module_store
    wasm_host
    storage
    proto_lib
    spdlog::spdlog
)

add_library(net_server STATIC ${NET_SRCS})
target_link_libraries(net_server
    proto_lib
    gRPC::grpc++
    Boost::system
    Boost::thread
    nlohmann_json::nlohmann_json
    spdlog::spdlog
)

add_library(changefeed_engine STATIC ${CHANGEFEED_SRCS})
target_link_libraries(changefeed_engine
    storage
    net_server
    spdlog::spdlog
)

add_library(admin_lib STATIC ${ADMIN_SRCS})
target_link_libraries(admin_lib
    module_store
    proto_lib
    spdlog::spdlog
)

# Main server executable
add_executable(instantdb src/cmd/server_main.cpp)

# Prototype demo executable
add_executable(instantdb_demo src/cmd/prototype_demo.cpp)
target_link_libraries(instantdb
    storage
    raft_consensus
    sql_engine
    wasm_host
    module_store
    net_server
    changefeed_engine
    admin_lib
    proto_lib
    Threads::Threads
    spdlog::spdlog
    fmt::fmt
)

target_link_libraries(instantdb_demo
    storage
    sql_engine
    changefeed_engine
    net_server
    Threads::Threads
    spdlog::spdlog
    fmt::fmt
)

# Tests
if(BUILD_TESTS)
    enable_testing()

    FetchContent_Declare(
        googletest
        GIT_REPOSITORY https://github.com/google/googletest.git
        GIT_TAG v1.14.0
    )
    FetchContent_MakeAvailable(googletest)

    file(GLOB_RECURSE TEST_SRCS "tests/*.cpp")
    add_executable(instantdb_tests ${TEST_SRCS})
    target_link_libraries(instantdb_tests
        storage
        raft_consensus
        sql_engine
        wasm_host
        module_store
        gtest_main
        gmock
    )

    add_test(NAME instantdb_tests COMMAND instantdb_tests)
endif()

# Installation
install(TARGETS instantdb DESTINATION bin)
install(DIRECTORY include/ DESTINATION include/instantdb)

# Package configuration
include(CPack)
set(CPACK_PACKAGE_NAME "InstantDB")
set(CPACK_PACKAGE_VERSION ${PROJECT_VERSION})
set(CPACK_PACKAGE_VENDOR "InstantDB")
set(CPACK_PACKAGE_DESCRIPTION_SUMMARY "In-memory SQL RDBMS with WASM modules")
